      'MONGODB_URI',
      'mongodb://admin:password@localhost:27017/octofit?authSource=admin'
  )
  
  client = MongoClient(MONGODB_URI)
  db = client['octofit']

  1. Check status: ./mongodb.sh status
  2. View logs: ./mongodb.sh logs
  3. Restart: ./mongodb.sh restart

If you can't connect from Django:
  1. Verify MongoDB is running: ./verify_mongodb.sh
  2. Check the connection string in Django settings
  3. Make sure username/password are correct
  4. Ensure the port is 27017
